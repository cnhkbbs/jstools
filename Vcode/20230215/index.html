import ctypes

headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/54.0.2840.99 Safari/537.36"}


def infomode2_retry(text, who):
    requests.packages.urllib3.disable_warnings(InsecureRequestWarning)
    MC = requests.get('https://api.ixiaowai.cn/mcapi/mcapi2.php', headers, verify=False, timeout=5).content
    with open('mc.png', 'wb') as fp:
        fp.write(MC)
    image = {
        'src': os.getcwd() + '\\mc.png',
        'placement': 'hero'
    }
    toast(text, "From:" + who, image=image)


def get_hitokoto(infomode):
    hitokoto = requests.get('https://v1.hitokoto.cn/', headers)
    hitokoto_text = json.loads(hitokoto.text)
    if infomode == 0:
        win32api.MessageBox(0, hitokoto_text["hitokoto"] + "\nfrom: " + str(hitokoto_text["from_who"]), "一言",
                            win32con.MB_OK)
    elif infomode == 1:
        toast(hitokoto_text["hitokoto"], "From:" + hitokoto_text["from_who"])
    else:
        if os.path.exists(os.getcwd() + '\\mc.png'):
            image = {
                'src': os.getcwd() + '\\mc.png',
                'placement': 'hero'
            }
            toast(hitokoto_text["hitokoto"], "From:" + hitokoto_text["from_who"], image=image)
        else:
            infomode2_retry(hitokoto_text["hitokoto"], hitokoto_text["from_who"])

        requests.packages.urllib3.disable_warnings(InsecureRequestWarning)
        MC = requests.get('https://api.ixiaowai.cn/mcapi/mcapi2.php', headers, verify=False, timeout=5).content
        with open('mc.png', 'wb') as fp:
            fp.write(MC)


def set_wallpaper():
    path = os.getcwd() + '\\photo.jpg'
    ctypes.windll.user32.SystemParametersInfoW(20, 0, path, 0)


def get_data(url, count):
    try:
        photo_json = requests.get(url, headers)  # 获取json数据
        photo_dict = json.loads(photo_json.text)  # 转换为字典
        randomnum = random.randint(0, count)
        target = photo_dict['data']['list'][randomnum]['url']
        photo = requests.get(target, headers).content
        with open('photo.jpg', 'wb') as fp:
            fp.write(photo)
        set_wallpaper()
    except Exception:
        win32api.MessageBox(0, "请检查您的网络是否通畅鸭", "啊哦！出错了", win32con.MB_OK)
